### Neo4j 图谱Schema

```cypher
-- 创建约束（唯一性）
CREATE CONSTRAINT unique_entity_name IF NOT EXISTS 
FOR (e:Entity) REQUIRE (e.name, e.type) IS UNIQUE;

-- 创建索引
CREATE INDEX entity_type_idx IF NOT EXISTS 
FOR (e:Entity) ON (e.type);

CREATE INDEX article_source_idx IF NOT EXISTS 
FOR (a:Article) ON (a.source);
```

**节点类型**：
```
(:Entity {name, type, aliases[], first_seen, confidence, neo4j_id})
(:Article {title, url, publish_time, source})
(:Report {title, created_at, report_type})
```

**关系类型**：
```
(article)-[:MENTIONS {count, confidence}]->(entity)
(entity)-[:RELATED_TO {relation_type, confidence, evidence}]->(entity)
(event)-[:HAPPENED_AT {time}]->(location)
(event)-[:INVOLVES {role}]->(participant)
(entity)-[:BELONGS_TO]->(topic)
(report)-[:DERIVED_FROM]->(article)
```

```python
# src/agents/state.py
from typing import List, Dict, Optional, Any
from dataclasses import dataclass, field
from pydantic import BaseModel
from enum import Enum

class TaskStatus(str, Enum):
    PENDING = "PENDING"
    RUNNING = "RUNNING"
    SUCCESS = "SUCCESS"
    FAILED = "FAILED"

class Entity(BaseModel):
    name: str
    type: str  # PER/ORG/LOC/TIME/EVENT/TOPIC
    confidence: float
    source: str
    first_seen: Optional[str] = None

class Relation(BaseModel):
    head: str
    tail: str
    relation_type: str
    confidence: float
    evidence: Optional[str] = None

class Document(BaseModel):
    id: str
    title: str
    content: str
    url: str
    source: str
    publish_time: str

@dataclass
class SharedState:
    """多Agent共享状态"""
    # 任务元信息
    task_id: str
    intent: str  # 用户意图
    status: TaskStatus = TaskStatus.PENDING
    
    # 任务规划
    plan: Optional[Dict[str, Any]] = None
    current_stage: str = "init"
    
    # 数据流
    raw_documents: List[Document] = field(default_factory=list)
    cleaned_documents: List[Document] = field(default_factory=list)
    extracted_entities: List[Entity] = field(default_factory=list)
    extracted_relations: List[Relation] = field(default_factory=list)
    
    # 质量元数据
    confidence_scores: Dict[str, float] = field(default_factory=dict)
    review_flags: List[Dict] = field(default_factory=list)
    
    # 最终产出
    report: Optional[str] = None
    knowledge_updates: Dict = field(default_factory=dict)
    
    # 持久化
    checkpoint_id: Optional[str] = None
    execution_log: List[Dict] = field(default_factory=list)
    
    def to_dict(self):
        """转换为字典用于序列化"""
        return {
            'task_id': self.task_id,
            'intent': self.intent,
            'status': self.status.value,
            'plan': self.plan,
            'current_stage': self.current_stage,
            'raw_documents': [doc.dict() for doc in self.raw_documents],
            'cleaned_documents': [doc.dict() for doc in self.cleaned_documents],
            'extracted_entities': [e.dict() for e in self.extracted_entities],
            'extracted_relations': [r.dict() for r in self.extracted_relations],
            'confidence_scores': self.confidence_scores,
            'review_flags': self.review_flags,
            'report': self.report,
            'knowledge_updates': self.knowledge_updates,
            'checkpoint_id': self.checkpoint_id,
            'execution_log': self.execution_log,
        }
```

### 2️⃣ Agent基类

```python
# src/agents/base.py
from typing import Callable, List, Dict, Any, Optional
from functools import wraps
from datetime import datetime
import logging
from shared_state import SharedState

logger = logging.getLogger(__name__)

class Tool:
    """工具定义"""
    def __init__(self, func: Callable, name: str, description: str, 
                 schema: Dict[str, Any], cost: str = "normal"):
        self.func = func
        self.name = name
        self.description = description
        self.schema = schema
        self.cost = cost  # "free" / "low" / "normal" / "high"
    
    async def __call__(self, **kwargs):
        try:
            result = await self.func(**kwargs) if hasattr(self.func, '__await__') else self.func(**kwargs)
            return {"status": "success", "data": result}
        except Exception as e:
            logger.error(f"Tool {self.name} failed: {e}")
            return {"status": "error", "error": str(e)}

def tool(name: str, description: str, schema: Dict = None, cost: str = "normal"):
    """工具装饰器"""
    def decorator(func: Callable):
        tool_obj = Tool(func, name, description, schema or {}, cost)
        func.tool = tool_obj
        return func
    return decorator

class BaseAgent:
    """Agent基类"""
    def __init__(self, name: str, role: str, goal: str, tools: List[Tool] = None):
        self.name = name
        self.role = role
        self.goal = goal
        self.tools = {t.name: t for t in (tools or [])}
        self.llm_client = None  # 由子类设置
        
    def register_tool(self, tool: Tool):
        """注册工具"""
        self.tools[tool.name] = tool
    
    def get_tool_schema(self) -> List[Dict]:
        """获取所有工具的OpenAI Function Calling Schema"""
        schemas = []
        for tool in self.tools.values():
            schemas.append({
                "type": "function",
                "function": {
                    "name": tool.name,
                    "description": tool.description,
                    "parameters": tool.schema,
                    "cost": tool.cost,
                }
            })
        return schemas
    
    def _log_action(self, state: SharedState, action: str, result: Dict):
        """记录执行日志"""
        state.execution_log.append({
            "timestamp": datetime.now().isoformat(),
            "agent": self.name,
            "action": action,
            "result": result,
        })
    
    async def run(self, state: SharedState) -> SharedState:
        """核心执行方法，子类覆盖"""
        raise NotImplementedError("Subclass must implement run()")
    
    def __repr__(self):
        return f"<{self.name} Agent: {self.role}>"
```

### 3️⃣ Collector Agent实现示例

```python
# src/agents/collector.py
from base import BaseAgent, Tool, tool
from shared_state import SharedState, Document
import feedparser
import httpx
from datetime import datetime
import hashlib
import asyncio

class CollectorAgent(BaseAgent):
    """采集管理Agent"""
    
    def __init__(self, llm_client):
        super().__init__(
            name="Collector",
            role="采集管理Agent",
            goal="根据任务调度采集多源媒体数据，执行去重与正文清洗",
        )
        self.llm_client = llm_client
        
        # 注册工具
        self.register_tool(self._create_rss_tool())
        self.register_tool(self._create_web_scraper_tool())
        self.register_tool(self._create_cleaner_tool())
    
    @staticmethod
    def _create_rss_tool() -> Tool:
        """创建RSS采集工具"""
        @tool(
            name="fetch_rss",
            description="从RSS源批量提取文章",
            schema={
                "type": "object",
                "properties": {
                    "rss_url": {"type": "string", "description": "RSS源URL"},
                    "limit": {"type": "integer", "description": "最多提取条数"}
                },
                "required": ["rss_url"]
            },
            cost="free"
        )
        async def fetch_rss(rss_url: str, limit: int = 10):
            feed = feedparser.parse(rss_url)
            docs = []
            for entry in feed.entries[:limit]:
                doc = Document(
                    id=hashlib.md5(entry.link.encode()).hexdigest(),
                    title=entry.title,
                    content=entry.get('summary', ''),
                    url=entry.link,
                    source="RSS",
                    publish_time=entry.get('published', datetime.now().isoformat()),
                )
                docs.append(doc)
            return {"articles": [d.dict() for d in docs], "count": len(docs)}
        
        return fetch_rss.tool
    
    @staticmethod
    def _create_web_scraper_tool() -> Tool:
        """创建Web爬虫工具"""
        @tool(
            name="scrape_web",
            description="爬取网页正文内容",
            schema={
                "type": "object",
                "properties": {
                    "url": {"type": "string"},
                    "selector": {"type": "string", "description": "CSS选择器（可选）"}
                },
                "required": ["url"]
            },
            cost="normal"
        )
        async def scrape_web(url: str, selector: str = "article"):
            # 实际实现使用 Scrapy + Playwright
            async with httpx.AsyncClient(timeout=10) as client:
                try:
                    response = await client.get(url)
                    # 简化版本，真实应用使用trafilatura或readability
                    return {
                        "url": url,
                        "title": response.headers.get('title', ''),
                        "content": response.text[:1000],
                        "status": "success"
                    }
                except Exception as e:
                    return {"status": "error", "error": str(e)}
        
        return scrape_web.tool
    
    @staticmethod
    def _create_cleaner_tool() -> Tool:
        """创建清洗工具"""
        @tool(
            name="clean_article",
            description="清洗文章正文，分离标题-正文-时间-来源",
            schema={
                "type": "object",
                "properties": {
                    "raw_html": {"type": "string"},
                    "url": {"type": "string"}
                },
                "required": ["raw_html"]
            },
            cost="free"
        )
        def clean_article(raw_html: str, url: str = ""):
            # 实际使用 trafilatura/readability
            # 这里演示简化逻辑
            import re
            text = re.sub(r'<[^>]+>', '', raw_html)
            return {
                "status": "success",
                "title": text[:100],
                "content": text,
                "cleaned_length": len(text)
            }
        
        return clean_article.tool
    
    async def run(self, state: SharedState) -> SharedState:
        """采集流程"""
        state.current_stage = "collection"
        
        # 解析用户意图中的数据源
        sources = self._parse_sources(state.intent)
        
        # 并行调用工具采集各源数据
        tasks = []
        for source in sources:
            if source.startswith("http"):
                tasks.append(self._collect_from_url(source))
            elif source.startswith("rss"):
                tasks.append(self._collect_from_rss(source))
        
        results = await asyncio.gather(*tasks)
        
        # 汇总数据
        for result in results:
            if result.get("status") == "success":
                for doc_dict in result.get("articles", []):
                    state.raw_documents.append(Document(**doc_dict))
        
        self._log_action(state, "fetch_raw_data", {
            "sources_count": len(sources),
            "articles_fetched": len(state.raw_documents)
        })
        
        return state
    
    async def _collect_from_rss(self, url: str) -> Dict:
        """采集RSS源"""
        result = await self.tools["fetch_rss"].func(url)
        return result
    
    async def _collect_from_url(self, url: str) -> Dict:
        """采集单个URL"""
        result = await self.tools["scrape_web"].func(url)
        cleaned = self.tools["clean_article"].func(result.get("content", ""))
        return {"status": "success", "articles": [{"title": result.get("title"),
                                                    "content": cleaned.get("content"),
                                                    "url": url,
                                                    "source": "Web",
                                                    "publish_time": datetime.now().isoformat()}]}
    
    def _parse_sources(self, intent: str) -> List[str]:
        """从意图中提取数据源"""
        # 实际应该使用LLM解析，这里简化
        return ["https://tech-news.com/feed", "https://example.com"]
```

```python
# src/agents/analyzer.py
from base import BaseAgent, Tool, tool
from shared_state import SharedState, Entity, Relation
import json
from typing import List

class AnalyzerAgent(BaseAgent):
    """分析抽取Agent"""
    
    def __init__(self, llm_client, small_model_ner=None):
        super().__init__(
            name="Analyzer",
            role="分析抽取Agent",
            goal="对清洗后的文本执行NLP信息抽取（NER、关系、事件、摘要）"
        )
        self.llm_client = llm_client
        self.small_model_ner = small_model_ner
        
        self.register_tool(self._create_ner_tool())
        self.register_tool(self._create_relation_tool())
        self.register_tool(self._create_summarizer_tool())
    
    def _create_ner_tool(self) -> Tool:
        """命名实体识别工具"""
        @tool(
            name="extract_entities",
            description="使用小模型进行命名实体识别",
            schema={
                "type": "object",
                "properties": {
                    "text": {"type": "string"},
                    "types": {"type": "array", "items": {"type": "string"}}
                },
                "required": ["text"]
            },
            cost="free"  # 小模型本地推理无成本
        )
        def extract_entities(text: str, types: List[str] = None):
            # 调用本地HanLP或BERT小模型
            if self.small_model_ner:
                entities = self.small_model_ner.predict(text)
            else:
                # Mock
                entities = [
                    {"text": "张三", "type": "PER", "confidence": 0.95},
                    {"text": "阿里巴巴", "type": "ORG", "confidence": 0.92},
                ]
            return {"entities": entities, "count": len(entities)}
        
        return extract_entities.tool
    
    def _create_relation_tool(self) -> Tool:
        """关系抽取工具"""
        @tool(
            name="extract_relations",
            description="使用LLM进行关系抽取",
            schema={
                "type": "object",
                "properties": {
                    "text": {"type": "string"},
                    "entities": {"type": "array"}
                },
                "required": ["text", "entities"]
            },
            cost="normal"
        )
        def extract_relations(text: str, entities: List[Dict]):
            # 调用LLM
            prompt = f"""
从以下文本中抽取主体-关系-宾语三元组。

文本：{text}

已识别实体：{json.dumps(entities, ensure_ascii=False)}

输出格式：
{{"relations": [
  {{"head": "实体1", "relation": "关系", "tail": "实体2", "confidence": 0.9}},
  ...
]}}
"""
            response = self.llm_client.call(prompt)
            try:
                return json.loads(response)
            except:
                return {"relations": []}
        
        return extract_relations.tool
    
    def _create_summarizer_tool(self) -> Tool:
        """摘要生成工具"""
        @tool(
            name="summarize",
            description="使用LLM生成文本摘要",
            schema={
                "type": "object",
                "properties": {
                    "text": {"type": "string"},
                    "max_length": {"type": "integer"}
                },
                "required": ["text"]
            },
            cost="normal"
        )
        def summarize(text: str, max_length: int = 200):
            prompt = f"将以下文本总结为不超过{max_length}字的摘要：\n{text}"
            summary = self.llm_client.call(prompt)
            return {"summary": summary, "length": len(summary)}
        
        return summarize.tool
    
    async def run(self, state: SharedState) -> SharedState:
        """分析流程"""
        state.current_stage = "analysis"
        
        # 遍历清洗后的文档
        for doc in state.cleaned_documents:
            # 第一步：实体识别（小模型，快速）
            ner_result = self.tools["extract_entities"].func(doc.content)
            entities = ner_result.get("entities", [])
            
            # 如有实体，执行关系抽取
            if entities:
                rel_result = self.tools["extract_relations"].func(
                    doc.content,
                    entities
                )
                for rel in rel_result.get("relations", []):
                    state.extracted_relations.append(
                        Relation(
                            head=rel["head"],
                            tail=rel["tail"],
                            relation_type=rel["relation"],
                            confidence=rel.get("confidence", 0.8),
                        )
                    )
            
            # 第三步：摘要生成
            summary_result = self.tools["summarize"].func(doc.content)
            
            # 存储实体
            for entity_text in entities:
                state.extracted_entities.append(
                    Entity(
                        name=entity_text["text"],
                        type=entity_text["type"],
                        confidence=entity_text.get("confidence", 0.8),
                        source=doc.source,
                    )
                )
        
        self._log_action(state, "extract_info", {
            "entities_count": len(state.extracted_entities),
            "relations_count": len(state.extracted_relations),
        })
        
        return state
```

```python
from langgraph.graph.graph import CompiledGraph
from shared_state import SharedState, TaskStatus
from typing import Dict, Any
import json

class GraphOrchestrator:
    """基于LangGraph的编排框架"""
    
    def __init__(self, agents: Dict[str, Any]):
        self.agents = agents
        self.graph = self._build_graph()
    
    def _build_graph(self) -> CompiledGraph:
        """构建编排DAG"""
        workflow = StateGraph(dict)  # 使用dict作为State类型
        
        # 添加节点
        workflow.add_node("plan", self._plan_node)
        workflow.add_node("collect", self._collect_node)
        workflow.add_node("analyze", self._analyze_node)
        workflow.add_node("model", self._model_node)
        workflow.add_node("review", self._review_node)
        workflow.add_node("report", self._report_node)
        workflow.add_node("correct", self._correct_node)
        
        # 添加边（控制流）
        workflow.add_edge(START, "plan")
        workflow.add_edge("plan", "collect")
        workflow.add_edge("collect", "analyze")
        workflow.add_edge("analyze", "model")
        workflow.add_edge("model", "review")
        
        # 条件路由
        workflow.add_conditional_edges(
            "review",
            self._review_decision,
            {
                "merge": "report",
                "correct": "correct",
            }
        )
        workflow.add_edge("correct", "review")
        workflow.add_edge("report", END)
        
        return workflow.compile()
    
    def _plan_node(self, state: Dict[str, Any]):
        """规划节点"""
        state_obj = SharedState(**state)
        state_obj = self.agents["planner"].run(state_obj)
        return state_obj.to_dict()
    
    def _collect_node(self, state: Dict[str, Any]):
        """采集节点"""
        state_obj = SharedState(**state)
        state_obj = asyncio.run(self.agents["collector"].run(state_obj))
        return state_obj.to_dict()
    
    def _analyze_node(self, state: Dict[str, Any]):
        """分析节点"""
        state_obj = SharedState(**state)
        state_obj = asyncio.run(self.agents["analyzer"].run(state_obj))
        return state_obj.to_dict()
    
    def _model_node(self, state: Dict[str, Any]):
        """知识建模节点"""
        state_obj = SharedState(**state)
        state_obj = asyncio.run(self.agents["knowledge_modeler"].run(state_obj))
        return state_obj.to_dict()
    
    def _review_node(self, state: Dict[str, Any]):
        """质检节点"""
        state_obj = SharedState(**state)
        state_obj = asyncio.run(self.agents["reviewer"].run(state_obj))
        return state_obj.to_dict()
    
    def _correct_node(self, state: Dict[str, Any]):
        """修正节点"""
        state_obj = SharedState(**state)
        # 根据review_flags进行回退修正
        state_obj = asyncio.run(self.agents["analyzer"].run(state_obj))
        return state_obj.to_dict()
    
    def _report_node(self, state: Dict[str, Any]):
        """报告生成节点"""
        state_obj = SharedState(**state)
        report = self._generate_report(state_obj)
        state_obj.report = report
        state_obj.status = TaskStatus.SUCCESS
        return state_obj.to_dict()
    
    def _review_decision(self, state: Dict[str, Any]) -> str:
        """质检决策：通过或修正"""
        state_obj = SharedState(**state)
        confidence = state_obj.confidence_scores.get("review", 0.0)
        
        if confidence >= 0.85 and not state_obj.review_flags:
            return "merge"
        else:
            return "correct"
    
    def _generate_report(self, state: SharedState) -> str:
        """生成分析报告"""
        report = f"""
# 分析报告

## 基本信息
- 任务ID: {state.task_id}
- 状态: {state.status.value}
- 采集文章数: {len(state.raw_documents)}
- 提取实体数: {len(state.extracted_entities)}
- 提取关系数: {len(state.extracted_relations)}

## 关键实体
{json.dumps([e.dict() for e in state.extracted_entities[:10]], ensure_ascii=False, indent=2)}

## 主要关系
{json.dumps([r.dict() for r in state.extracted_relations[:10]], ensure_ascii=False, indent=2)}

## 质量评估
- 平均置信度: {sum(state.confidence_scores.values()) / len(state.confidence_scores) if state.confidence_scores else 0:.2%}
"""
        return report
    
    async def run(self, task_id: str, intent: str):
        """执行完整流程"""
        initial_state = SharedState(
            task_id=task_id,
            intent=intent,
            status=TaskStatus.PENDING,
        )
        
        result = self.graph.invoke(initial_state.to_dict())
        return SharedState(**result)
```

## 五、关键技术难点与解决方案总结

| 难点 | 方案 |
|------|------|
| **Agent状态一致性** | LangGraph StateGraph单一状态源，原子更新 |
| **任务DAG动态调整** | Planner可被重新触发，生成降级计划 |
| **实体对齐准确性** | 三级漏斗：向量初筛→图结构→LLM仲裁 |
| **LLM成本控制** | 分层策略：小模型粗筛+LLM精抽 |
| **并发控制** | asyncio + StateGraph原子写入 |
| **断点恢复** | State定期MySQL持久化 |
